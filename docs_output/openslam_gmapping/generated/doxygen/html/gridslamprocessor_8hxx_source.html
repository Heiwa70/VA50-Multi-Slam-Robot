42inline void GridSlamProcessor::normalize(){
43 //normalize the log m_weights
44 double gain=1./(m_obsSigmaGain*m_particles.size());
45 double lmax= -std::numeric_limits<double>::max();
46 for (ParticleVector::iterator it=m_particles.begin(); it!=m_particles.end(); it++){
47 lmax=it->weight>lmax?it->weight:lmax;
48 }
49 //cout << "!!!!!!!!!!! maxwaight= "<< lmax << endl;
50
51 m_weights.clear();
52 double wcum=0;
53 m_neff=0;
54 for (std::vector<Particle>::iterator it=m_particles.begin(); it!=m_particles.end(); it++){
55 m_weights.push_back(exp(gain*(it->weight-lmax)));
56 wcum+=m_weights.back();
57 //cout << "l=" << it->weight<< endl;
58 }
59
60 m_neff=0;
61 for (std::vector<double>::iterator it=m_weights.begin(); it!=m_weights.end(); it++){
62 *it=*it/wcum;
63 double w=*it;
64 m_neff+=w*w;
65 }
66 m_neff=1./m_neff;
67
68}

70inline bool GridSlamProcessor::resample(const double* plainReading, int adaptSize, const RangeReading* reading){
71
72 bool hasResampled = false;
73
74 TNodeVector oldGeneration;
75 for (unsigned int i=0; i<m_particles.size(); i++){
76 oldGeneration.push_back(m_particles[i].node);
77 }
78
79 if (m_neff<m_resampleThreshold*m_particles.size()){
80
81 if (m_infoStream)
82 m_infoStream << "*************RESAMPLE***************" << std::endl;
83
84 uniform_resampler<double, double> resampler;
85 m_indexes=resampler.resampleIndexes(m_weights, adaptSize);
86
87 if (m_outputStream.is_open()){
88 m_outputStream << "RESAMPLE "<< m_indexes.size() << " ";
89 for (std::vector<unsigned int>::const_iterator it=m_indexes.begin(); it!=m_indexes.end(); it++){
90 m_outputStream << *it << " ";
91 }
92 m_outputStream << std::endl;
93 }
94
95 onResampleUpdate();
96 //BEGIN: BUILDING TREE
97 ParticleVector temp;
98 unsigned int j=0;
99 std::vector<unsigned int> deletedParticles; //this is for deleteing the particles which have been resampled away.
100
101 // cerr << "Existing Nodes:" ;
102 for (unsigned int i=0; i<m_indexes.size(); i++){
103 // cerr << " " << m_indexes[i];
104 while(j<m_indexes[i]){
105 deletedParticles.push_back(j);
106 j++;
107 }
108 if (j==m_indexes[i])
109 j++;
110 Particle & p=m_particles[m_indexes[i]];
111 TNode* node=0;
112 TNode* oldNode=oldGeneration[m_indexes[i]];
113 // cerr << i << "->" << m_indexes[i] << "B("<<oldNode->childs <<") ";
114 node=new TNode(p.pose, 0, oldNode, 0);
115 //node->reading=0;
116 node->reading=reading;
117 // cerr << "A("<<node->parent->childs <<") " <<endl;
118
119 temp.push_back(p);
120 temp.back().node=node;
121 temp.back().previousIndex=m_indexes[i];
122 }
123 while(j<m_indexes.size()){
124 deletedParticles.push_back(j);
125 j++;
126 }
127 // cerr << endl;
128 std::cerr << "Deleting Nodes:";
129 for (unsigned int i=0; i<deletedParticles.size(); i++){
130 std::cerr <<" " << deletedParticles[i];
131 delete m_particles[deletedParticles[i]].node;
132 m_particles[deletedParticles[i]].node=0;
133 }
134 std::cerr << " Done" <<std::endl;
135
136 //END: BUILDING TREE
137 std::cerr << "Deleting old particles..." ;
138 m_particles.clear();
139 std::cerr << "Done" << std::endl;
140 std::cerr << "Copying Particles and Registering scans...";
141 for (ParticleVector::iterator it=temp.begin(); it!=temp.end(); it++){
142 it->setWeight(0);
143 m_matcher.invalidateActiveArea();
144 m_matcher.registerScan(it->map, it->pose, plainReading);
145 m_particles.push_back(*it);
146 }
147 std::cerr << " Done" <<std::endl;
148 hasResampled = true;
149 } else {
150 int index=0;
151 std::cerr << "Registering Scans:";
152 TNodeVector::iterator node_it=oldGeneration.begin();
153 for (ParticleVector::iterator it=m_particles.begin(); it!=m_particles.end(); it++){
154 //create a new node in the particle tree and add it to the old tree
155 //BEGIN: BUILDING TREE
156 TNode* node=0;
157 node=new TNode(it->pose, 0.0, *node_it, 0);
158
159 //node->reading=0;
160 node->reading=reading;
161 it->node=node;
162
163 //END: BUILDING TREE
164 m_matcher.invalidateActiveArea();
165 m_matcher.registerScan(it->map, it->pose, plainReading);
166 it->previousIndex=index;
167 index++;
168 node_it++;
169
170 }
171 std::cerr << "Done" <<std::endl;
172
173 }
174 //END: BUILDING TREE
175
176 return hasResampled;
177}
