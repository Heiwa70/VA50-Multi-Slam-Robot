65void orientedpoint<T,A>::normalize() {
66 if (theta >= -M_PI && theta < M_PI)
67 return;
68
69 int multiplier = (int)(theta / (2*M_PI));
70 theta = theta - multiplier*2*M_PI;
71 if (theta >= M_PI)
72 theta -= 2*M_PI;
73 if (theta < -M_PI)
74 theta += 2*M_PI;

169inline orientedpoint<T,A>
170interpolate(const orientedpoint<T,A>& p1, const F& t1, const orientedpoint<T,A>& p2, const F& t2, const F& t3){
171 F gain=(t3-t1)/(t2-t1);
172 orientedpoint<T,A> p;
173 p.x=p1.x+(p2.x-p1.x)*gain;
174 p.y=p1.y+(p2.y-p1.y)*gain;
175 double s=sin(p1.theta)+sin(p2.theta)*gain,
